import React from 'react'
import { connect } from 'react-redux'
import * as actions from '../actions'
 
const getBlocks = (tetris, piece, ghost, pieceColor) => {
  const blocks = [];
  tetris.forEach((col, i) => {
    col.forEach((color, j) => {
      if (piece && piece.filter(([x, y]) => x==i && y==j).length)
        color = pieceColor
      if (ghost && ghost.filter(([x, y]) => x==i && y==j).length && color === '')
        color = 'ghost ' + pieceColor
      blocks.push(<div key={`${i}${j}`} className={`block ${color}`}></div>);
    });
  });
  return blocks
}
 
const App = ({ tetris, startGame, message, ghost, coords, color }) => {
  return (
    <div className='gameContainer'>
      {tetris !== null ?
      <div className='tetrisContainer'>
        {getBlocks(tetris, coords, ghost, color)}
      </div> :
      <button onClick={startGame}>{message}</button>}
    </div>
  );
}
 
const mapDispatchToProps = (dispatch) => {
  return {
    startGame: () => {
      dispatch(actions.startGame())
      dispatch(actions.newPiece())
    },
  }
}
 
const mapStateToProps = (state) => {
  return {
    tetris: state.tetris,
    message: state.message,
    ghost: state.ghost,
    coords: state.coords,
    color: state.color,
  }
}
 
export default connect(mapStateToProps, mapDispatchToProps)(App)
